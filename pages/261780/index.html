    判断页面各种距离

    scrollTop：一个元素的内容垂直滚动的像素数
    offsetHeight：包括元素的 CSS height + CSS padding +边框+ 水平滚动条高度 (如果存在)
    ，不包含:before或:after等伪类元素的高度。
    **clientHeight：**这个属性是只读属性，对于没有定义CSS或者内联布局盒子的元素为0，否则，它是元素内部的高度(单位像素)，包含内边距，但不包括水平滚动条、边框和外边距。
    scrollHeight ：属性是一个只读属性，它返回该元素的像素高度，高度包含内边距（padding），不包含外边距（margin）、边框（border），是一个整数，单位是像素 px。

    # 判断到底部

    mounted(){
  let chat_scroll = this.$refs['chat_content' + this.userid]
  chat_scroll.addEventListener('scroll',this.scrollBotton)
},
//离开页面后销毁
destroyed(){
  chat_scroll.removeEventListener('scroll',this.scrollBotton)
},
methods:{
  // 监听是否回到底部
  scrollBotton(){
    let chat_scroll = this.$refs['chat_content' + this.userid]
    let scrollTop = chat_scroll.scrollTop
    let clientHeight = chat_scroll.clientHeight
    let scrollHeight = chat_scroll.scrollHeight
    if(scrollTop + clientHeight >= scrollHeight - 20){
      window.clearInterval(this.b_color_time1);
      window.clearInterval(this.b_color_time2);
      this.bottom_color='#969696'
      console.log('到底啦')
    }
   },
 }
